       ```ts
       import { PrismaClient } from '@prisma/client';
       import { normalizeForDb } from './tag-normalize';

       export function createTagNormalizationExtension<T extends PrismaClient>(prisma: T) {
         return prisma.$extends({
           name: 'cameraTagNormalization',
           query: {
             camera: {
               async create({ args, query }) {
                 if (Array.isArray((args.data as any)?.tags)) {
                   (args.data as any).tagsNormalized = normalizeForDb((args.data as any).tags);
                 }
                 return query(args);
               },
               async update({ args, query }) {
                 if (Array.isArray((args.data as any)?.tags)) {
                   (args.data as any).tagsNormalized = normalizeForDb((args.data as any).tags);
                 }
                 return query(args);
               },
               async upsert({ args, query }) {
                 if (Array.isArray((args.create as any)?.tags)) {
                   (args.create as any).tagsNormalized = normalizeForDb((args.create as any).tags);
                 }
                 if (Array.isArray((args.update as any)?.tags)) {
                   (args.update as any).tagsNormalized = normalizeForDb((args.update as any).tags);
                 }
                 return query(args);
               },
             },
           },
         });
       }
       ```
       NOTE: `updateMany` and `createMany` are intentionally NOT hooked — bulk paths use per-row `update` calls (per Pitfall 5 + D-12), so they always trigger the per-row `update` hook. Bulk import (D-10) goes through `create` (per row) which is hooked above.
