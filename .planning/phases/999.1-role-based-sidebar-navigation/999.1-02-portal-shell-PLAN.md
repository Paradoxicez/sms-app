       ```typescript
       "use client";
       import { useEffect, useState } from "react";
       import { authClient } from "@/lib/auth-client";
       import { apiFetch } from "@/lib/api";

       export type UserRole = "admin" | "user";
       export type MemberRole = "admin" | "operator" | "developer" | "viewer";

       export function useCurrentRole() {
         const [state, setState] = useState<{
           userRole: UserRole | null;
           memberRole: MemberRole | null;
           activeOrgId: string | null;
           activeOrgName: string | null;
           loading: boolean;
         }>({ userRole: null, memberRole: null, activeOrgId: null, activeOrgName: null, loading: true });

         useEffect(() => {
           let cancelled = false;
           (async () => {
             const s = await authClient.getSession();
             if (!s.data?.user) { if (!cancelled) setState({ userRole: null, memberRole: null, activeOrgId: null, activeOrgName: null, loading: false }); return; }
             const userRole = s.data.user.role as UserRole;
             const activeOrgId = s.data.session?.activeOrganizationId ?? null;
             let memberRole: MemberRole | null = null;
             let activeOrgName: string | null = null;
             if (userRole === "user" && activeOrgId) {
               try {
                 const m = await apiFetch<{ role: MemberRole }>(`/api/organizations/${activeOrgId}/members/me`);
                 memberRole = m.role;
               } catch { memberRole = null; }
               try {
                 const orgs = await authClient.organization.list();
                 const match = orgs.data?.find((o: { id: string; name: string }) => o.id === activeOrgId);
                 activeOrgName = match?.name ?? null;
               } catch { /* noop */ }
             }
             if (!cancelled) setState({ userRole, memberRole, activeOrgId, activeOrgName, loading: false });
           })();
           return () => { cancelled = true; };
         }, []);

         return state;
       }
       ```

       ```typescript
       "use client";
       import { LayoutDashboard, MapPin, Camera, FolderTree, Building2, SlidersHorizontal, Film, ShieldCheck, UsersRound, FileText, Key, Bell } from "lucide-react";
       import { NavShell, NavRow, NavGroup } from "./nav-shell";
       import { useFeatures } from "@/hooks/use-features";
       import type { MemberRole } from "@/hooks/use-current-role";

       interface Props {
         userName?: string;
         userEmail?: string;
         memberRole: MemberRole;
         activeOrgId: string;
         activeOrgName: string;
       }

       // Role → permitted hrefs (D-11). admin sees all.
       const ROLE_MATRIX: Record<MemberRole, ReadonlyArray<string> | "ALL"> = {
         admin: "ALL",
         operator:  ["/app/dashboard","/app/cameras","/app/map","/app/recordings","/app/audit-log"],
         developer: ["/app/dashboard","/app/cameras","/app/map","/app/developer/api-keys","/app/developer/webhooks","/app/audit-log"],
         viewer:    ["/app/dashboard","/app/cameras","/app/map","/app/recordings","/app/audit-log"],
       };

       interface Item { label: string; href: string; icon: React.ComponentType<{className?:string}>; feature?: keyof ReturnType<typeof useFeatures>["features"] | string; }
       const OVERVIEW: Item[] = [
         { label: "Dashboard", href: "/app/dashboard", icon: LayoutDashboard },
         { label: "Map",       href: "/app/map",        icon: MapPin, feature: "map" },
       ];
       const CAMERAS: Item[] = [
         { label: "Cameras",         href: "/app/cameras",         icon: Camera },
         { label: "Projects",        href: "/app/projects",        icon: FolderTree },
         { label: "Sites",           href: "/app/sites",           icon: Building2 },
         { label: "Stream Profiles", href: "/app/stream-profiles", icon: SlidersHorizontal },
         { label: "Recordings",      href: "/app/recordings",      icon: Film, feature: "recordings" },
         { label: "Policies",        href: "/app/policies",        icon: ShieldCheck },
       ];
       const ORGANIZATION: Item[] = [
         { label: "Team",      href: "/app/team",      icon: UsersRound },
         { label: "Audit Log", href: "/app/audit-log", icon: FileText, feature: "auditLog" },
       ];
       const DEVELOPER: Item[] = [
         { label: "API Keys", href: "/app/developer/api-keys", icon: Key, feature: "apiKeys" },
         { label: "Webhooks", href: "/app/developer/webhooks", icon: Bell, feature: "webhooks" },
       ];

       export function TenantNav({ userName, userEmail, memberRole, activeOrgId, activeOrgName }: Props) {
         const { isEnabled } = useFeatures(activeOrgId);
         const permitted = ROLE_MATRIX[memberRole];
         const allow = (item: Item) => {
           if (permitted !== "ALL" && !permitted.includes(item.href)) return false;
           if (item.feature && !isEnabled(item.feature)) return false;
           return true;
         };
         const truncate = (n: string) => n.length > 16 ? n.slice(0,16) + "…" : n;
         return (
           <NavShell badgeLabel={truncate(activeOrgName || "Workspace")} badgeTitle={activeOrgName || undefined} userName={userName} userEmail={userEmail}>
             <NavGroup caption="Overview" firstGroup>
               {OVERVIEW.filter(allow).map((i) => <NavRow key={i.href} {...i} exactMatch={i.href === "/app/dashboard"} />)}
             </NavGroup>
             <NavGroup caption="Cameras">
               {CAMERAS.filter(allow).map((i) => <NavRow key={i.href} {...i} />)}
             </NavGroup>
             <NavGroup caption="Organization">
               {ORGANIZATION.filter(allow).map((i) => <NavRow key={i.href} {...i} />)}
             </NavGroup>
             <NavGroup caption="Developer">
               {DEVELOPER.filter(allow).map((i) => <NavRow key={i.href} {...i} />)}
             </NavGroup>
           </NavShell>
         );
       }
       ```
       Note: the Overview group MUST pass `firstGroup` to NavGroup — this suppresses the leading `<Separator />` that NavGroup otherwise renders above the caption. Without it, a stray divider would appear above "Overview".
