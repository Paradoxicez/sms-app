   ```bash
   log "[1/6] Pre-backup snapshot"
   declare -A PRE_COUNTS
   for table in User Organization Member Camera Recording; do
     PRE_COUNTS["${table}"]=$(pg_count "${table}")
     log "  Pre  ${table}: ${PRE_COUNTS[${table}]}"
   done

   # Validate starter data — fail fast if any table is empty (verify-playback didn't populate)
   if [[ "${PRE_COUNTS[User]}" -lt 1 ]] || [[ "${PRE_COUNTS[Organization]}" -lt 1 ]] || \
      [[ "${PRE_COUNTS[Member]}" -lt 1 ]] || [[ "${PRE_COUNTS[Camera]}" -lt 1 ]]; then
     die "Insufficient starter data — User=${PRE_COUNTS[User]} Org=${PRE_COUNTS[Organization]} Member=${PRE_COUNTS[Member]} Camera=${PRE_COUNTS[Camera]}. Run bootstrap.sh + register a camera first."
     exit 2
   fi
   if [[ "${PRE_COUNTS[Recording]}" -lt 1 ]]; then
     warn "  Recording count = 0 — verify-playback's record-60s step did not produce a Recording row?"
     warn "  Continuing — backup will still test other entities."
   fi

   # Snapshot MinIO bucket listings (sorted, for deterministic comparison)
   ${DC} exec -T minio mc alias set local http://localhost:9000 "${MINIO_ROOT_USER:-minioadmin}" "${MINIO_ROOT_PASSWORD}" >/dev/null 2>&1 || true
   local pre_avatars pre_recordings pre_snapshots
   pre_avatars=$(${DC} exec -T minio mc ls --recursive local/avatars/ 2>/dev/null | awk '{print $NF}' | sort | sha256sum | awk '{print $1}')
   pre_recordings=$(${DC} exec -T minio mc ls --recursive local/recordings/ 2>/dev/null | awk '{print $NF}' | sort | sha256sum | awk '{print $1}')
   pre_snapshots=$(${DC} exec -T minio mc ls --recursive local/snapshots/ 2>/dev/null | awk '{print $NF}' | sort | sha256sum | awk '{print $1}')
   log "  Pre  avatars sha256:    ${pre_avatars}"
   log "  Pre  recordings sha256: ${pre_recordings}"
   log "  Pre  snapshots sha256:  ${pre_snapshots}"
   echo
   ```
   Note: We sha256 the SORTED file-name listing as a stable digest. mc's metadata (timestamps, sizes) drifts non-deterministically; only the file-key set is contractually preserved.
