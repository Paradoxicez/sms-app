model AuditLog {
  id         String   @id @default(uuid())
  orgId      String
  userId     String?
  action     String   // "create" | "update" | "delete"
  resource   String   // "camera" | "project" | "site" | "policy" | "streamProfile" | "apiKey" | "webhook" | "user" | "organization" | "settings"

  @@index([orgId, createdAt])
  @@index([orgId, userId])
  @@index([orgId, resource])
  @@index([orgId, action])
}

model Notification {
  id        String   @id @default(uuid())
  orgId     String
  userId    String
  type      String   // "camera.online" | "camera.offline" | "camera.degraded" | "camera.reconnecting" | "system.alert"
  title     String
  body      String?
  data      Json?    // { cameraId, previousStatus, etc. }
  read      Boolean  @default(false)
  createdAt DateTime @default(now())

  @@index([userId, read, createdAt])
  @@index([orgId, createdAt])
}

model NotificationPreference {
  id        String   @id @default(uuid())
  userId    String
  orgId     String
  eventType String   // "camera.online" | "camera.offline" | "camera.degraded" | "camera.reconnecting" | "system.alert"
  enabled   Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([userId, orgId, eventType])
}

**audit.interceptor.ts**: NestJS interceptor implementing `NestInterceptor`. Per D-14:
- Only intercepts POST, PUT, PATCH, DELETE methods
- Skips paths starting with `/api/srs/callbacks` (SRS internal callbacks) and `/api/health`
- Gets userId from `request.user?.id`, orgId from `this.cls.get('ORG_ID')`
- Derives action: POST=create, PUT/PATCH=update, DELETE=delete
- Derives resource from path: `/api/cameras/*` -> "camera", `/api/projects/*` -> "project", `/api/sites/*` -> "site", `/api/policies/*` -> "policy", `/api/stream-profiles/*` -> "streamProfile", `/api/api-keys/*` -> "apiKey", `/api/webhooks/*` -> "webhook", `/api/users/*` -> "user", `/api/organizations/*` -> "organization", `/api/settings/*` -> "settings". Default: extract second path segment.
- Gets resourceId from response data `responseData?.id` or `request.params?.id`
- Gets IP from `request.ip || request.headers['x-forwarded-for']`
- In `tap()` operator, calls `auditService.log(...)` with `.catch(() => {})` — fire-and-forget, never block response
- Per D-14: strips password/secret/token fields from details before passing to service
