```tsx
async function handleImport() {
  setImporting(true);
  try {
    // Client-side skip duplicates (server also dedupes per P04)
    const payloadRows = rows.filter((r) => Object.keys(r.errors).length === 0 && !r.duplicate);
    const response = await fetch('/api/cameras/bulk-import', {
      method: 'POST',
      credentials: 'include',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ cameras: payloadRows.map(mapRowToPayload), siteId: selectedSiteId }),
    });
    if (!response.ok) throw new Error(`Import failed: ${response.status}`);
    const result = await response.json();
    const imported = result.imported ?? 0;
    const skipped = result.skipped ?? 0;

    if (imported > 0 && skipped === 0) {
      toast.success(`Imported ${imported} cameras successfully.`);
    } else if (imported > 0 && skipped > 0) {
      toast.success(`Imported ${imported} cameras, skipped ${skipped} duplicates.`);
    } else if (imported === 0 && skipped > 0) {
      toast.warning(`No cameras imported — all ${skipped} rows were duplicates.`);
    } else {
      toast.error('Import failed. Check camera limits and try again.');
    }

    onImportComplete?.(); // existing callback, adjust to actual name
  } catch (err) {
    toast.error('Import failed. Check camera limits and try again.');
  } finally {
    setImporting(false);
  }
}
```
