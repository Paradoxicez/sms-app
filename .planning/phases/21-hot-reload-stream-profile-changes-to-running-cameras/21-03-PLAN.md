Replace the existing `updateCamera(id, dto)` method (lines 288-297) with this exact body:

```typescript
async updateCamera(
  id: string,
  dto: UpdateCameraDto,
  triggeredBy: { userId: string; userEmail: string } | { system: true } = {
    system: true,
  },
): Promise<any> {
  // Load pre-image WITH the existing profile so we can fingerprint the old.
  const pre = await this.tenancy.camera.findUnique({
    where: { id },
    include: { streamProfile: true },
  });
  if (!pre) {
    throw new NotFoundException(`Camera ${id} not found`);
  }

  // Phase 19.1 D-01: ingestMode is immutable post-create.
  const safe: any = { ...dto };
  delete safe.ingestMode;

  const updated = await this.tenancy.camera.update({
    where: { id },
    data: safe,
    include: { streamProfile: true },
  });

  // Phase 21 D-02: profile reassign detection.
  let restartTriggered = false;

  // Only consider a restart when streamProfileId is in the PATCH body
  // AND the new value differs from the old.
  const profileChanged =
    Object.prototype.hasOwnProperty.call(dto, 'streamProfileId') &&
    dto.streamProfileId !== pre.streamProfileId;

  if (profileChanged) {
    // Load the new profile (or null if dto.streamProfileId is null).
    const newProfile = updated.streamProfile;
    const oldFp = fingerprintProfile(pre.streamProfile);
    const newFp = fingerprintProfile(newProfile);

    if (oldFp !== newFp) {
      // Eligibility gate (D-02 + RESEARCH §2 maintenance/status filter).
      const eligible =
        ['online', 'connecting', 'reconnecting', 'degraded'].includes(
          pre.status,
        ) && pre.maintenanceMode === false;

      if (eligible && this.streamsService) {
        const result = await this.streamsService.enqueueProfileRestart({
          profileId: dto.streamProfileId ?? 'none-sentinel',
          oldFingerprint: oldFp,
          newFingerprint: newFp,
          triggeredBy,
          originPath: `/api/cameras/${id}`,
          originMethod: 'PATCH',
          cameraId: id,
        });
        restartTriggered = result.affectedCameras > 0;
      }
    }
  }

  return { ...updated, restartTriggered };
}
```

```typescript
@Patch('cameras/:id')
async updateCamera(
  @Param('id') id: string,
  @Body() body: unknown,
  @Req() req: Request,
) {
  const result = UpdateCameraSchema.safeParse(body);
  if (!result.success) {
    throw new BadRequestException(result.error.flatten());
  }
  const user = (req as any).user;
  const triggeredBy =
    user?.id && user?.email
      ? { userId: user.id, userEmail: user.email }
      : ({ system: true } as const);
  return this.camerasService.updateCamera(id, result.data, triggeredBy);
}
```
