```bash
#!/usr/bin/env bash
# ============================================================================
# deploy/scripts/bootstrap.sh — Phase 29 (DEPLOY-18)
#
# First-run orchestrator. Brings a fresh VM from "deploy/.env filled" to
# "operator can log in at https://${DOMAIN}" in under 10 minutes.
#
# Usage:
#   cp deploy/.env.production.example deploy/.env
#   $EDITOR deploy/.env   # set DOMAIN, ADMIN_EMAIL, GHCR_ORG, ACME_EMAIL
#   bash deploy/scripts/bootstrap.sh
#
# Idempotent: safe to re-run after Ctrl-C, partial failure, or successful
# completion (re-run rotates ADMIN_PASSWORD if .env changed; otherwise no-op).
# ============================================================================

set -euo pipefail
IFS=$'\n\t'

# --- Path resolution (works from any CWD) ---
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
DEPLOY_DIR="$(cd "${SCRIPT_DIR}/.." && pwd)"
ENV_FILE="${DEPLOY_DIR}/.env"
COMPOSE_FILE="${DEPLOY_DIR}/docker-compose.yml"

# --- Color helpers (TTY-aware, D-29) ---
if [[ -t 1 ]] && [[ "$(tput colors 2>/dev/null || echo 0)" -ge 8 ]]; then
  GREEN=$'\033[32m'; RED=$'\033[31m'; YELLOW=$'\033[33m'; BOLD=$'\033[1m'; RESET=$'\033[0m'
else
  GREEN=''; RED=''; YELLOW=''; BOLD=''; RESET=''
fi
log()  { printf '%s[bootstrap]%s %s\n' "${BOLD}" "${RESET}" "$*"; }
ok()   { printf '%s[bootstrap] ✓%s %s\n' "${GREEN}" "${RESET}" "$*"; }
warn() { printf '%s[bootstrap] ⚠%s %s\n' "${YELLOW}" "${RESET}" "$*" >&2; }
die()  { printf '%s[bootstrap] ✗%s %s\n' "${RED}" "${RESET}" "$*" >&2; exit 1; }

START=$(date +%s)

# --- D-07: Pre-flight (3 checks, <10s total) ---
log "Pre-flight checks..."
docker info >/dev/null 2>&1 || die "Docker daemon not running. Start Docker first."
[[ -f "${ENV_FILE}" ]] || die "${ENV_FILE} missing. Run: cp ${DEPLOY_DIR}/.env.production.example ${ENV_FILE}, edit, then re-run bootstrap.sh."
grep -qE '^DOMAIN=.+' "${ENV_FILE}" || die "DOMAIN= not set in ${ENV_FILE}. Edit it then re-run bootstrap.sh."
ok "Pre-flight passed"

# --- D-08: Auto-secrets if placeholders detected ---
if grep -qE '^[A-Z_]+=change-me-' "${ENV_FILE}" || grep -qE '^[A-Z_]+=$' "${ENV_FILE}"; then
  log "Generating secrets via init-secrets.sh..."
  bash "${SCRIPT_DIR}/init-secrets.sh"
else
  log "All secrets already set (init-secrets.sh skipped)"
fi

# --- Source .env so we can read ADMIN_EMAIL / ADMIN_PASSWORD / DOMAIN below ---
# shellcheck disable=SC1090
set -a; source "${ENV_FILE}"; set +a

# Validate the operator-supplied identifiers init-secrets does NOT generate (D-07 #3 + manual fields).
[[ -n "${ADMIN_EMAIL:-}" ]]    || die "ADMIN_EMAIL is empty in ${ENV_FILE}. Edit it (operator-supplied identifier; init-secrets cannot generate)."
[[ -n "${ADMIN_PASSWORD:-}" ]] || die "ADMIN_PASSWORD is empty in ${ENV_FILE}. Re-run init-secrets.sh or set it manually."
[[ -n "${GHCR_ORG:-}" ]]       || die "GHCR_ORG is empty in ${ENV_FILE}. Set it to the github org that owns the GHCR images."
[[ -n "${ACME_EMAIL:-}" ]]     || warn "ACME_EMAIL empty — Caddy will register an anonymous account. Cert renewal warnings will not reach you. See deploy/DOMAIN-SETUP.md."

DC="docker compose -f ${COMPOSE_FILE} --env-file ${ENV_FILE}"

# --- Pull images (idempotent — docker layer cache) ---
log "Pulling images for IMAGE_TAG=${IMAGE_TAG:-latest}..."
${DC} pull
ok "Images pulled"

# --- D-09: Wait sms-migrate to exit 0 BEFORE create-admin ---
log "Running migrate + bucket init + stream-profile seed (sms-migrate)..."
${DC} up -d --wait sms-migrate || die "sms-migrate failed. Inspect: ${DC} logs sms-migrate"
ok "sms-migrate exited 0 (schema + buckets + default profile ready)"

# --- D-09: Bring up api so we can exec bin/sms (api healthcheck waits on sms-migrate already) ---
log "Starting api + web + caddy..."
${DC} up -d
log "Waiting for api healthcheck..."
# Wait up to 60s for api healthy (FFmpeg + Prisma + Better Auth init).
# Use Go template form (stable across compose v2.x minor versions; JSON parsing
# of `ps --format json` has been observed to drift between v2.10..v2.20).
for i in $(seq 1 12); do
  STATE=$(${DC} ps --format '{{.Health}}' api 2>/dev/null | head -1 || true)
  if [[ "${STATE}" == "healthy" ]]; then break; fi
  sleep 5
done

# --- D-09: create-admin with idempotent --force fallback ---
log "Creating super-admin ${ADMIN_EMAIL}..."
if ${DC} exec -T api bin/sms create-admin --email "${ADMIN_EMAIL}" --password "${ADMIN_PASSWORD}" 2>/tmp/bootstrap-create-admin.err; then
  ok "Super-admin created"
elif grep -q 'already exists' /tmp/bootstrap-create-admin.err 2>/dev/null; then
  log "Super-admin exists — rotating password via --force..."
  ${DC} exec -T api bin/sms create-admin --email "${ADMIN_EMAIL}" --password "${ADMIN_PASSWORD}" --force \
    || die "create-admin --force failed. Inspect: ${DC} logs api"
  ok "Super-admin password rotated"
else
  cat /tmp/bootstrap-create-admin.err >&2
  die "create-admin failed. Inspect: ${DC} logs api"
fi
rm -f /tmp/bootstrap-create-admin.err

# --- D-10: Wait HTTPS reachable (poll Caddy /api/health 5s × 24 = 120s) ---
log "Waiting for HTTPS endpoint (Caddy provisioning Let's Encrypt cert, ~30-60s typical)..."
HTTPS_READY=0
for i in $(seq 1 24); do
  if curl -fsS -o /dev/null --max-time 5 "https://${DOMAIN}/api/health" 2>/dev/null; then
    HTTPS_READY=1
    break
  fi
  sleep 5
done

if [[ "${HTTPS_READY}" -eq 1 ]]; then
  ok "HTTPS ready"
else
  warn "HTTPS not reachable after 120s. Caddy may still be issuing cert."
  warn "Check: ${DC} logs caddy"
  warn "DNS A-record correct? Port 80 reachable from Internet? See deploy/DOMAIN-SETUP.md."
fi

# --- D-12: Final summary + timing log ---
ELAPSED=$(( $(date +%s) - START ))
echo
ok "Stack live at https://${DOMAIN}"
log "  Login email:    ${ADMIN_EMAIL}"
log "  Bootstrap time: ${ELAPSED}s"
log "  Day-2 ops:"
log "    Update:  bash deploy/scripts/update.sh <new-tag>"
log "    Backup:  bash deploy/scripts/backup.sh"
log "    Restore: bash deploy/scripts/restore.sh <archive>"
```

**Critical correctness notes (do NOT deviate):**

1. **`set -euo pipefail` + `IFS=$'\n\t'`** — D-29 mandate. Without `-u` a typo in `${ADMIN_PASSWROD}` silently passes empty string to bin/sms; without `-o pipefail` a piped failure goes undetected.

2. **Source the .env file** with `set -a; source ENV_FILE; set +a` so values become env vars without manual export. The `set -a` toggle auto-exports every variable assigned during the source. SC1090 disable comment is required because shellcheck cannot statically resolve a dynamic source path.

3. **`docker compose ... ps --format '{{.Health}}'`** — health probe between `up -d` and create-admin. The api healthcheck (Phase 25 line 113) is `curl -fsS http://localhost:3003/api/health`; we wait up to 60s before invoking `compose exec api`. Don't omit this loop — `compose up -d` returns as soon as containers are CREATED, not HEALTHY, and `compose exec api` fails if api hasn't bound port 3003 yet. The Go template form (`--format '{{.Health}}'`) is stable across compose v2.x minor versions; the JSON form (`--format json` + grep) has drifted between v2.10..v2.20.

4. **`/tmp/bootstrap-create-admin.err`** captures stderr so we can grep for "already exists" without re-running create-admin (which would race the --force path). The temp file is removed on the success branch and overwritten on failure (no leak).

5. **HTTPS poll uses `curl --max-time 5`** — without max-time a hung Caddy process during ACME challenge could block the loop indefinitely. 24 iterations × max 5s/iter = 120s upper bound matches D-10.

6. **`die` on sms-migrate failure, `warn` on HTTPS timeout** — sms-migrate failure is unrecoverable (schema not applied; api will crash-loop). HTTPS timeout is recoverable (cert may issue 30-60s after our 120s budget; operator just refreshes browser). D-10 explicitly lists this as warn-only.

7. **chmod +x the script** in source tree:
```bash
chmod +x deploy/scripts/bootstrap.sh
git update-index --chmod=+x deploy/scripts/bootstrap.sh
```
Same rationale as Plan 29-01 Task 2 — without git-index 100755 the script ships into a fresh `git clone` non-executable.

8. **No `if true; then ... fi` workarounds, no `|| true` swallowing.** Every command's failure is either fatal (`die`) or warned (`warn`). The single exception: `STATE=$(${DC} ps --format '{{.Health}}' api ... || true)` — that's intentional because `ps` returns non-zero when a service hasn't been created yet (during the loop's first iteration if api boot is slow), and we want to retry the loop, not exit.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| operator shell → bootstrap.sh stdin | Script reads ADMIN_PASSWORD from .env (chmod 600 by init-secrets) and passes via argv to `bin/sms create-admin`. |
| bootstrap.sh → docker compose exec | The password traverses `compose exec -T api bin/sms create-admin --password "${ADMIN_PASSWORD}"` — visible in process listings on the host for the lifetime of the exec (typically <1s). |
| .env file → host filesystem | chmod 600 by init-secrets ensures only the operator user can read. If operator runs bootstrap as root, .env is root:root mode 600 — still acceptable for self-hosted single-server. |
| /tmp/bootstrap-create-admin.err | Stderr captured to a temp file in /tmp; readable by other host users with the right permissions. Does NOT contain the password (only the create-admin error message). Removed on success path. |
