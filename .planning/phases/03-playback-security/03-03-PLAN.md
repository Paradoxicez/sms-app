2. **Embed player page** (`apps/web/src/app/embed/[session]/page.tsx`):
```typescript
'use client';
import { useEffect, useRef, useState } from 'react';
import { useParams } from 'next/navigation';
import Hls from 'hls.js';

interface SessionInfo {
  id: string;
  hlsUrl: string;
  expiresAt: string;
  cameraId: string;
}

export default function EmbedPage() {
  const params = useParams();
  const sessionId = params.session as string;
  const videoRef = useRef<HTMLVideoElement>(null);
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function init() {
      try {
        const API_BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:3003';
        const res = await fetch(`${API_BASE}/api/playback/sessions/${sessionId}`);
        if (!res.ok) {
          if (res.status === 404) {
            setError('Session not found. The playback session may have expired or been revoked.');
          } else {
            setError('Unable to load video stream. Please check your connection and try again.');
          }
          setLoading(false);
          return;
        }

        const session: SessionInfo = await res.json();

        // Check if expired
        if (new Date(session.expiresAt) < new Date()) {
          setError('This playback session has expired. Request a new session from the API to continue viewing.');
          setLoading(false);
          return;
        }

        if (!videoRef.current) return;

        if (Hls.isSupported()) {
          const hls = new Hls({
            xhrSetup: (xhr) => {
              // hls.js will use the token from the m3u8 URL for segment/key fetches
            },
          });
          hls.loadSource(session.hlsUrl);
          hls.attachMedia(videoRef.current);
          hls.on(Hls.Events.MANIFEST_PARSED, () => {
            videoRef.current?.play().catch(() => {});
            setLoading(false);
          });
          hls.on(Hls.Events.ERROR, (_, data) => {
            if (data.fatal) {
              if (data.type === Hls.ErrorTypes.NETWORK_ERROR) {
                setError('Stream is currently offline. The camera may be disconnected or the stream has not been started.');
              } else {
                setError('Unable to load video stream. Please check your connection and try again.');
              }
            }
          });

          return () => hls.destroy();
        } else if (videoRef.current.canPlayType('application/vnd.apple.mpegurl')) {
          // Safari native HLS
          videoRef.current.src = session.hlsUrl;
          videoRef.current.addEventListener('loadedmetadata', () => {
            videoRef.current?.play().catch(() => {});
            setLoading(false);
          });
        } else {
          setError('Your browser does not support HLS playback.');
          setLoading(false);
        }
      } catch {
        setError('Unable to load video stream. Please check your connection and try again.');
        setLoading(false);
      }
    }

    init();
  }, [sessionId]);

  return (
    <div style={{
      width: '100vw',
      height: '100vh',
      background: '#000',
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      overflow: 'hidden',
    }}>
      {loading && !error && (
        <div style={{ color: '#fff' }}>
          {/* Loader2 spinner or simple CSS spinner */}
          <svg className="animate-spin" width="32" height="32" viewBox="0 0 24 24" fill="none" stroke="white" strokeWidth="2">
            <path d="M21 12a9 9 0 11-6.219-8.56" />
          </svg>
        </div>
      )}

      {error && (
        <p style={{
          color: 'hsl(0, 0%, 70%)',
          fontSize: '16px',
          textAlign: 'center',
          padding: '20px',
          maxWidth: '400px',
        }}>
          {error}
        </p>
      )}

      <video
        ref={videoRef}
        style={{
          width: '100%',
          height: '100%',
          display: error ? 'none' : 'block',
        }}
        controls
        autoPlay
        muted
        playsInline
      />
    </div>
  );
}
```
