enum PolicyLevel {
  SYSTEM
  PROJECT

model Policy {
  id              String       @id @default(uuid())
  orgId           String?      // null for SYSTEM level
  level           PolicyLevel
  name            String
  description     String?

  ttlSeconds      Int?         // Session TTL in seconds (null = inherit)
  maxViewers      Int?         // Max concurrent viewers, 0 = unlimited (null = inherit)
  domains         String[]     @default([])  // Domain allowlist, empty = allow all
  allowNoReferer  Boolean?     // Allow requests without Referer (null = inherit)
  rateLimit       Int?         // Per-API-key requests per minute (null = inherit)

  cameraId        String?      @unique
  camera          Camera?      @relation(fields: [cameraId], references: [id], onDelete: Cascade)
  siteId          String?      @unique
  site            Site?        @relation(fields: [siteId], references: [id], onDelete: Cascade)
  projectId       String?      @unique
  project         Project?     @relation(fields: [projectId], references: [id], onDelete: Cascade)

  createdAt       DateTime     @default(now())
  updatedAt       DateTime     @updatedAt

  @@index([orgId])
  @@index([level])
}

model PlaybackSession {
  id              String       @id @default(uuid())
  orgId           String
  cameraId        String
  camera          Camera       @relation(fields: [cameraId], references: [id], onDelete: Cascade)
  token           String       @unique
  hlsUrl          String

  ttlSeconds      Int
  maxViewers      Int
  domains         String[]     @default([])
  allowNoReferer  Boolean      @default(true)

  expiresAt       DateTime
  createdAt       DateTime     @default(now())

  @@index([orgId])
  @@index([cameraId])
  @@index([token])
  @@index([expiresAt])
}

  - `JWT_PLAYBACK_SECRET` from `process.env.JWT_PLAYBACK_SECRET` -- encode with `new TextEncoder().encode(secret)`. Fallback to a generated random secret in dev (log warning if using fallback).
  - `createSession(cameraId: string, orgId: string)`:
    1. Verify camera exists and belongs to org
    2. Call `policiesService.resolve(cameraId)` to get resolved policy
    3. Check viewer limit: `statusService.getViewerCount(cameraId)` against `resolved.maxViewers`. If maxViewers > 0 and currentViewers >= maxViewers, throw ForbiddenException with message "Viewer limit reached ({current}/{max})"
    4. Create PlaybackSession record in DB with resolved policy snapshot fields (ttlSeconds, maxViewers, domains, allowNoReferer)
    5. Sign JWT with jose:
       ```typescript
       const token = await new SignJWT({
         cam: cameraId,
         org: orgId,
         domains: resolved.domains,
       })
         .setProtectedHeader({ alg: 'HS256' })
         .setSubject(session.id)
         .setIssuedAt()
         .setExpirationTime(`${resolved.ttlSeconds}s`)
         .sign(this.jwtSecret);
       ```
    6. Compute hlsUrl: `http://srs:8080/live/${orgId}/${cameraId}.m3u8?token=${token}` (internal SRS URL -- external proxy handled separately)

    1. Call `jwtVerify(token, this.jwtSecret)`
    2. Check payload.cam === cameraId and payload.org === orgId
    3. Look up PlaybackSession by payload.sub (sessionId)
    4. Return session data with resolved policy fields, or null if invalid

  - `getSession(sessionId: string)`:
    1. Find PlaybackSession by id
    2. Return session with expiresAt check (return null if expired)

  - `matchDomain(pageUrl: string, allowedDomains: string[], allowNoReferer: boolean): boolean`:
    1. If !pageUrl or empty: return allowNoReferer (per D-16)
    2. If allowedDomains.length === 0: return true (per D-14)
    3. Parse hostname from pageUrl with try/catch (malformed = treat as no referer)
    4. Check each pattern: '*' matches all, '*.example.com' matches hostname ending with '.example.com' or exact match 'example.com', exact match (per D-15)

- `PlaybackController`:
  - `POST /api/cameras/:cameraId/sessions` (AuthGuard) -- calls createSession, returns { sessionId, hlsUrl, expiresAt }
  - `GET /api/playback/sessions/:id` (NO AuthGuard -- public for embed page) -- calls getSession, returns { id, hlsUrl, expiresAt, cameraId } or 404

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-03-01 | Spoofing | PlaybackController POST /sessions | mitigate | AuthGuard validates session before session creation; org context from CLS |

| T-03-03 | Information Disclosure | GET /playback/sessions/:id | accept | Returns minimal data (id, hlsUrl, expiresAt); no sensitive policy details exposed. Session IDs are UUIDs (unguessable) |
| T-03-04 | Elevation of Privilege | Policy CRUD | mitigate | AuthGuard + org-scoped queries via CLS; system-level policies only editable by super admin |
| T-03-05 | Denial of Service | Session creation | mitigate | Viewer concurrency limit enforced at session creation; rate limiting added in Plan 02 |
</threat_model>
