       ```ts
       async bulkTagAction(
         orgId: string,
         triggeredBy: { userId: string; userEmail?: string },
         dto: BulkTagsDto,
       ): Promise<{ updatedCount: number }> {
         const target = dto.tag.trim();
         // DTO validates non-empty + max length already; double-check defensively:
         if (!target) throw new BadRequestException('Tag must not be empty');

         const cameras = await this.tenancy.camera.findMany({
           where: { id: { in: dto.cameraIds } },
           select: { id: true, tags: true, orgId: true },
         });
         // RLS reduces this set to caller's org automatically.

         let updatedCount = 0;
         for (const cam of cameras) {
           const before = cam.tags ?? [];
           let after: string[];
           if (dto.action === 'add') {
             const lowerSet = new Set(before.map((t) => t.toLowerCase()));
             if (lowerSet.has(target.toLowerCase())) continue; // idempotent dedup
             after = normalizeForDisplay([...before, target]);
           } else {
             // remove
             const lower = target.toLowerCase();
             after = before.filter((t) => t.toLowerCase() !== lower);
             if (after.length === before.length) continue; // no-op
           }

           await this.tenancy.camera.update({
             where: { id: cam.id },
             data: { tags: after },
             // tagsNormalized auto-updated by camera-tag.extension (Plan 22-01)
           });
           updatedCount += 1;

           // D-26: per-camera audit row with diff
           await this.auditService.log({
             orgId,
             userId: triggeredBy.userId,
             action: 'update',
             resource: 'camera',
             resourceId: cam.id,
             method: 'POST',
             path: '/api/cameras/bulk/tags',
             details: {
               bulkAction: dto.action,
               tag: target,
               diff: { tags: { before, after } },
             },
           });
         }

         // Invalidate distinct-tags cache so the new tag appears in autocomplete next call
         await this.tagCacheService.invalidate(orgId);

         return { updatedCount };
       }
       ```
