   const HTML_ESCAPES: Record<string, string> = {
     '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;',
   };
   export function escapeHtml(str: string | null | undefined): string {
     if (str == null) return '';
     return String(str).replace(/[&<>"']/g, (ch) => HTML_ESCAPES[ch]);
   }
