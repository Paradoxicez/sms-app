```typescript
export const TENANCY_CLIENT = Symbol('TENANCY_CLIENT');
export function createTenancyExtension(
  prisma: PrismaClient,
  cls: ClsService,
): ReturnType<PrismaClient['$extends']>;
// Extension wraps ONLY $allModels.$allOperations. It emits (inside a
// prisma.$transaction):
//   SELECT set_config('app.current_org_id', <ORG_ID>,     TRUE)  — if cls.get('ORG_ID')
//   SELECT set_config('app.is_superuser',  'true',         TRUE)  — if cls.get('IS_SUPERUSER')
// It does NOT cover $queryRaw, $executeRaw, or outer $transaction wrappers.
// `TRUE` = is_local → scope is CURRENT TRANSACTION ONLY, no leak across queries.
```
